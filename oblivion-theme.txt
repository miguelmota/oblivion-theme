	<script>
	$(document).ready(function(){

		// Like button. Send 'like' url to hidden iframe
		$('.like-button').click(function(){
			var id = $(this).attr('data-id');
		    var oath = $(this).attr('data-url').slice(-8);
		    var like = 'http://www.tumblr.com/like/'+oath+'?id='+id;
			$('#likedit').css('display','block').attr('src', like).css('display','none');
			$(this).after('<span class="icon icon-heart-red-16 icon-no-opacity like-button"></span>').remove();
		})

		// Navigation hover effect
		$('.main-side-nav-ul a:not(".selected")').live({
				mouseenter:
					function(){
						$(this).animate({borderLeftWidth: '6px'}, {queue: false, duration: 60});
					},
				mouseleave:
					function(){
						$(this).animate({borderLeftWidth: '4px'}, {queue: false, duration: 60});
					}
			});
	
		// Add selected class to nav link based on page name
		$('.main-side-nav-ul a').each(function(){ 
			if($(this).attr('id') == $('.main-content-wrap').attr('class').split(' ')[1]){
				$('#home').removeClass('selected');
				$(this).addClass('selected');
				$('.icon', this).addClass('icon-no-opacity');
			}
			else if(location.pathname == $(this).attr('href')){
				$('#home').removeClass('selected');
				$(this).addClass('selected');
				$('.icon', this).addClass('icon-no-opacity');
			}
		});

		// Highlight icon on link hover
		$('a:not(".selected")').has('.icon').live({
			mouseenter:
				function(){
					$('.icon', this).addClass('icon-no-opacity');
				},
			mouseleave:
				function(){
					$('.icon', this).removeClass('icon-no-opacity');
				}
		});
			
		// Format Tweet date to timeago
		$('.latest-tweet-date').timeago();

		// Likes carousel
		var speed = 5000;
		var run = setInterval('rotate()', speed);
		var item_width = $('#likes li').outerWidth();
		var left_value = item_width * (-1);
		$('#likes li:first').before($('#likes li:last'));
		$('#likes').css({'left':left_value});
		$('.likes-prev').live('click', function(){
			var left_indent = parseInt($('#likes').css('left')) - item_width;
			$('#likes').animate({'left':left_indent}, 200, function(){
				$('#likes li:last').after($('#likes li:first'));
				$('#likes').css({'left':left_value});
			});
			return false;
		});
		$('.likes-next').live('click', function(){
			var left_indent = parseInt($('#likes').css('left')) + item_width;
			$('#likes').animate({'left':left_indent}, 200, function(){
				$('#likes li:first').before($('#likes li:last'));
				$('#likes').css({'left':left_value});
			});
			return false;
		});
		$('#likes').live({
			mouseenter:
				function(){
					clearInterval(run);
					},
			mouseleave:
				function(){
					run = setInterval('rotate()', speed);
				}
		});
				
		// Show zoom icon on image hover
		$('.photo-wrap').live({
			mouseenter:
				function(){
					$('.full-res-link', this).fadeIn('fast');
				},
			mouseleave:
				function(){
					$('.full-res-link', this).fadeOut('fast');
				}
		});

		// Following expand more
		$('.following-expand').toggle(function(){
				$('.following-ul').css('max-height', '96px');
				$(this).text('less');
			},
			function(){
				$('.following-ul').css('max-height', '24px');
				$(this).text('more');
			}
		);

		// Highlight short url on focus
		$('.short-url').focus(function(){
			$(this).select();
		});
		$('.short-url').mouseup(function(e){
	        e.preventDefault();
		});
		
		// Change default text color on input focus
		$('input:text:not(".short-url"), input:password, textarea').focus(function(){
			if(this.value === this.defaultValue){
				this.value = '';
			}
			$(this).css({
				color: '#999',
				fontStyle: 'normal'
			});
		}).blur(function(){
			if(this.value === ''){
				this.value = this.defaultValue;
				$(this).css({
				color: '#555',
				fontStyle: 'italic'
			});
			}
		});
	
		// Open external links in new tab
		$('a[rel=external]').live('click', function(){
			window.open(this.href);
			return false;
		});

		// Back to top smooth scroll effect
		$('a[href*=#]').click(function() {
			if (location.pathname.replace(/^\//,'') == this.pathname.replace(/^\//,'')
			&& location.hostname == this.hostname) {
	            var $target = $(this.hash);
	            $target = $target.length && $target || $('[name=' + this.hash.slice(1) +']');
	            if ($target.length) {
	            	var targetOffset = $target.offset().top;
	            	$('html,body').animate({scrollTop: targetOffset}, 1000);
	                return false;
	            }
	        }
	    });
		$(window).scroll(function () { 
			var scrollTop = $(document).scrollTop();
			scrollTop = parseInt(scrollTop);
			var offset = topYloc+scrollTop+'px';  
			$('.top-link').animate({top:offset},{duration:500,queue:false});
		});
		topYloc = parseInt($('.top-link').css('top').substring(0,$('.top-link').css('top').indexOf('px')));

	});
		
	$(window).scroll(function(){
		// Toggle back to top button on scroll
		if(window.pageYOffset >= 200){
			$('.top-link').fadeIn(600);
		}
		if(window.pageYOffset < 200){
			$('.top-link').fadeOut(185);
		}
	});

	// Likes rotate carousel
	function rotate() {
		$('.likes-next').click();
	}
	
	// Display the year
	function displayYear(){
		var date = new Date();
		var thisYear = date.getFullYear();
	
		document.write(thisYear);
	}

		<script>
		var fixed = false;
		$(window).scroll(function(){
			
			// Main side nav scroll to fixed
			if(window.innerWidth >= 769) {

				if($(this).scrollTop() >= 150){
					if(!fixed){
						fixed = true;
						$('.site-description,.social-links,.likes-wrap,.following-wrap').hide();
						$('.main-side-nav').css({
							position: 'fixed'
						});
					}
				}
				else {
					if(fixed){
						fixed = false;
						$('.site-description,.social-links,.likes-wrap,.following-wrap').show();
						$('.main-side-nav').css({
							position: 'static'
						});
					}
				}

			}
			else {
				if($(this).scrollTop() >= 150){
					if(!fixed){
						fixed = true;
						$('.main-side-nav').css({
							position: 'static'
						});
					}
				}
				else {
					if(fixed){
						fixed = false;
						$('.main-side-nav').css({
							position: 'static'
						});
					}
				}
				
			}
		});
		</script>
